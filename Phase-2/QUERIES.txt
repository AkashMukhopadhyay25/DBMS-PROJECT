SELECT ORIGIN, DESTINATION, FARE, DEPARTURE TIME FROM ROUTE, BUS WHERE BUS.ROUTE_ID = ROUTE.ID AND ORIGIN = "DELHI";
SELECT DESTINATION,FARE,  BUS.ID FROM (BUS JOIN ROUTE ON BUS.ROUTE_ID = ROUTE.ID) WHERE AC = 1 AND ORIGIN LIKE "B%";
SELECT SEAT, TIMESTAMP, DATE FROM RESERVE, PASSENGER WHERE RESERVE.PASSENGER_ID = PASSENGER.ID AND NAME = "SHUBHAM";
SELECT BUS.ID, AC , FARE , NAME, EMAIL , TIMESTAMP FROM BUS, PASSENGER, RESERVE WHERE RESERVE.BUS_ID = BUS.ID AND RESERVE.PASSENGER_ID = PASSENGER.ID;
SELECT ORIGIN, DESTINATION , ARRIVAL_TIME , DEPARTURE_TIME, FARE FROM ROUTE, BUS WHERE ROUTE.ID = BUS.ROUTE_ID AND FARE BETWEEN 20 AND 60 AND BUS.ID>4; 
SELECT RESERVE.ID , SEAT , NAME FROM PASSENGER, RESERVE WHERE PASSENGER_ID = PASSENGER.ID AND SEAT>30;
